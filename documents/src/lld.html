3. Logic Layer (app/logic) — algorithms

3.1 series.py

@dataclass DailySeries: dates: list[date], values: np.ndarray   # aligned, gap-filled
load_daily_f107(records) -> DailySeries         # from SWPC f107_cm_flux json
kp_to_daily(records) -> (DailySeries max_kp, DailySeries ap)    # 3-hourly -> daily
    # Ap from Kp via the standard ap-from-Kp lookup (Kp thirds -> ap: 0..400)
align(*series) -> common daily grid; forward/interp gap-fill <= 3-day gaps
trailing_mean(values, window) -> np.ndarray     # 27-day rotation mean etc.

3.2 f107_forecast.py

point_forecast(history: DailySeries, lead_days: int) -> np.ndarray
    # x0 = last observed; m27 = trailing 27-day mean; mcyc = trailing 90-day mean
    # reversion weight w(L) = 1 - exp(-L / TAU), TAU ~ 10 days
    # target(L) = blend(m27, mcyc, by L)         # short lead -> 27d, long -> cycle
    # f(L) = x0 + (target(L) - x0) * w(L)         # persistence -> reversion
    # floored at settings.f107_floor_sfu

quantile_bands(point: np.ndarray, err_quantiles: dict[int, dict[float,float]])
    -> dict[str, np.ndarray]                      # p05,p25,p50,p75,p95 per lead
    # band[q][L] = point[L] + err_quantiles[L][q]; enforce monotone; floor

3.3 kp_forecast.py

CATEGORIES = ["quiet","unsettled","active","storm"]   # Kp <3, 3-4, 4-5, >=5
category_probabilities(daily_kp: DailySeries, lead_days: int, halflife=180)
    -> list[dict[str,float]]
    # current state s0 = category(last max_kp)
    # from history build recency-weighted transition counts state->category@lead
    # P(cat | s0, L) = weighted_count / total ; Laplace-smoothed; sums to 1
expected_ap(daily_ap, lead_days) -> (point, lo, hi)   # persistence + climatology band

3.4 calibration.py — verification core

walk_forward_errors(history, lead_days, min_train, step)
    -> dict[int, np.ndarray]                      # signed errors per lead (OOS)
    # for origin t in [min_train .. N-lead) step k:
    #     train = history[:t+1]                    # STRICT — no future (CON-2)
    #     pred  = point_forecast(train, lead)
    #     err[L].append(history[t+L] - pred[L-1])
    # returns per-lead error arrays -> quantiles feed f107 bands

error_quantiles(errors, qs=(5,25,50,75,95)) -> dict[int, dict[float,float]]
coverage(history, bands_by_origin) -> dict[int, {p50_50, p90}]   # empirical coverage
pinball_loss(y, q_pred, q) -> float                               # proper score
skill_vs_reference(pen_rmse, noaa_rmse) -> 1 - pen/noaa           # per lead
reliability(pred_probs, outcomes, bins=10) -> [(pred, obs, n)]    # Kp reliability

3.5 drag.py

density_at(alt_km, f107) -> rho                   # reduced model:
    # rho(h,F) = rho0(h) * exp((F - F_ref)/F_scale)   # F10.7-scaled exponential atm.
    # rho0(h): tabulated reference density; F_ref=150, F_scale documented
density_band(alt_km, f107_bands) -> {p05,p50,p95}
decay_uncertainty(alt_km, f107_bands, bstar_like, days)
    -> along-track_km {p05,p50,p95}               # da/dt ~ -k*rho*v ; integrate band
    # labelled reduced-model, order-of-magnitude (FR-DRAG-3)

6. Service Layer Detail

6.1 swpc_client.py

BASE = "https://services.swpc.noaa.gov"
fetch_f107()        -> json    /json/f107_cm_flux.json
fetch_kp()          -> json    /products/noaa-planetary-k-index.json
fetch_noaa45()      -> json    /json/45-day-forecast.json
fetch_history()     -> json    /json/solar-cycle/observed-solar-cycle-indices.json
# httpx, timeout 30 s, 2 retries exp backoff; failures -> DataSourceError

6.2 forecast_service.py — pipeline (sequence)

refresh():                                        # startup + scheduled
  1 pipeline.stage("swpc_sync")     -> cache.get_or_fetch(each source, loader|demo)
  2 pipeline.stage("history")       -> series.load + kp_to_daily + align + gapfill
  3 pipeline.stage("backtest")      -> walk_forward_errors + error_quantiles + Kp stats
  4 pipeline.stage("forecast")      -> f107 point+bands, kp categories, expected Ap
  5 pipeline.stage("calibration")   -> coverage, pinball, skill vs NOAA, reliability
  6 pipeline.stage("publish")       -> persist snapshot; expose read-model
  each stage: try/except -> record status/items/error; degrade to last-known-good;
  the pipeline never aborts the app (FR-ING-6).

6.3 briefing_service.py

If anthropic_api_key set: Claude Messages API (claude-opus-4-8, thinking:{type:"adaptive"}, max_tokens 400, 10 s timeout) summarising the current forecast + calibration into an operator note; on any failure log WARN and return a deterministic template. Never raises to caller.
